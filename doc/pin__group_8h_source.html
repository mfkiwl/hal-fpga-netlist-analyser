324  Result<std::monostate> move_pin(T* pin, u32 new_index)
325  {
326  if (pin == nullptr)
327  {
328  return ERR("'nullptr' given instead of a pin when trying to move pin within pin group '" + m_name + "' with ID " + std::to_string(m_id));
329  }
330 
331  if (pin->m_group.first != this)
332  {
333  return ERR("pin '" + pin->get_name() + "' with ID " + std::to_string(pin->get_id()) + " does not belong to pin group '" + m_name + "' with ID " + std::to_string(m_id));
334  }
335 
336  if (new_index >= m_start_index && new_index < m_next_index)
337  {
338  u32 old_index = pin->m_group.second;
339  if (old_index == new_index)
340  {
341  return OK({});
342  }
343 
344  i32 direction = (old_index > new_index) ? 1 : -1;
345  m_pins.erase(std::next(m_pins.begin(), old_index - m_start_index));
346  auto it = std::next(m_pins.begin(), new_index - m_start_index);
347  it = m_pins.insert(it, pin);
348  for (u32 i = new_index; i != old_index; i += direction)
349  {
350  std::advance(it, direction);
351  std::get<1>((*it)->m_group) += direction;
352  }
353  std::get<1>(pin->m_group) = new_index;
354 
355  return OK({});
356  }
357  else
358  {
359  return ERR("new index (" + std::to_string(new_index) + ") for pin '" + pin->get_name() + "' with ID " + std::to_string(pin->get_id()) + " is not between start index ("
360  + std::to_string(m_start_index) + ") and end index(" + std::to_string(m_next_index - 1) + ") of pin group '" + m_name + "' with ID " + std::to_string(m_id));
361  }
362  }
