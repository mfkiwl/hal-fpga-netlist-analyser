213  template<typename T>
214  CORE_API std::vector<T> split(const T& s, const char delim, bool obey_brackets = false)
215  {
216  std::vector<T> result;
217  T item = "";
218 
219  if (obey_brackets)
220  {
221  int bracket_level = 0;
222 
223  for (size_t i = 0; i < s.length(); ++i)
224  {
225  char c = s.at(i);
226  switch (c)
227  {
228  case '(':
229  case '{':
230  case '[':
231  ++bracket_level;
232  break;
233  case ')':
234  case '}':
235  case ']':
236  --bracket_level;
237  break;
238  default:
239  break;
240  }
241  if (bracket_level < 0)
242  {
243  bracket_level = 0;
244  }
245  if (c == delim)
246  {
247  // No constant expression, therefore not usable in switch case
248  if (bracket_level == 0)
249  {
250  result.push_back(item);
251  item = "";
252  }
253  else
254  {
255  item.push_back(c);
256  }
257  }
258  else
259  {
260  item.push_back(c);
261  }
262  }
263  if (!item.empty())
264  {
265  result.push_back(item);
266  }
267  }
268  else
269  {
270  std::stringstream ss(s);
271  while (std::getline(ss, item, delim))
272  {
273  result.push_back(item);
274  }
275  }
276  if (s.back() == delim)
277  {
278  result.push_back("");
279  }
280  return result;
281  }

360  {
361  auto s = str;
362  size_t pos = 0;
363 
364  if (search.empty())

366  return str;
367  }
368 
369  while ((pos = s.find(search, pos)) != T::npos)
370  {
371  s.replace(pos, search.length(), replace);
372  pos += replace.length();
373  }
374 
375  return s;
