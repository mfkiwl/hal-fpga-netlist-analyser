561  GraphContext* GraphContextManager::restoreFromFile(const QString& filename)
562  {
563  QFile jsFile(filename);
564  if (!jsFile.open(QIODevice::ReadOnly))
565  return nullptr;
566  QJsonDocument jsonDoc = QJsonDocument::fromJson(jsFile.readAll());
567  const QJsonObject& json = jsonDoc.object();
568  GraphContext* firstContext = nullptr;
569  GraphContext* selectedContext = nullptr;
570 
571  if (json.contains("views") && json["views"].isArray())
572  {
573  QJsonArray jsonViews = json["views"].toArray();
574  int nviews = jsonViews.size();
575  for (int iview = 0; iview < nviews; iview++)
576  {
577  QJsonObject jsonView = jsonViews.at(iview).toObject();
578  if (!jsonView.contains("id") || !jsonView["id"].isDouble())
579  continue;
580  u32 viewId = jsonView["id"].toInt();
581  if (!jsonView.contains("name") || !jsonView["name"].isString())
582  continue;
583  QString viewName = jsonView["name"].toString();
584  if (viewId > mMaxContextId)
585  mMaxContextId = viewId;
586  int visibleFlag = 1; // default to visible before flag was invented
587  if (jsonView.contains("visible"))
588  visibleFlag = jsonView["visible"].toInt();
589  GraphContext* context = gGraphContextManager->getContextById(viewId);
590  if (context)
591  {
592  // load view in existing context
593  context->clear();
594  if (viewName != context->mName)
595  renameGraphContextAction(context,viewName);
596  if (!context->readFromFile(jsonView))
597  {
598  log_warning("gui", "failed to set up existing context ID={} from view file {}.", context->id(), filename.toStdString());
599  continue;
600  }
601  }
602  else
603  {
604  // create new context
605  context = new GraphContext(viewId, viewName);
606  context->setLayouter(getDefaultLayouter(context));
607  context->setShader(getDefaultShader(context));
608  // context->setExclusiveModuleId(exclusiveModuleId); TODO exclusive module ID
609  context->scene()->setDebugGridEnabled(mSettingDebugGrid->value().toBool());
610  connect(mSettingDebugGrid, &SettingsItemCheckbox::boolChanged, context->scene(), &GraphicsScene::setDebugGridEnabled);
611 
612  if (!context->readFromFile(jsonView))
613  {
614  log_warning("gui", "failed to create context context ID={} view file {}.", context->id(), filename.toStdString());
615  deleteGraphContext(context);
616  continue;
617  }
618 
619  mContextTableModel->beginInsertContext(context);
620  mContextTableModel->addContext(context);
621  mContextTableModel->endInsertContext();
622  if (visibleFlag)
623  Q_EMIT contextCreated(context);
624  }
625 
626  if (jsonView.contains("exclusiveModuleId"))
627  context->setExclusiveModuleId(jsonView["exclusiveModuleId"].toInt(),false);
628  if (jsonView.contains("selected"))
629  selectedContext = context;
630  if (visibleFlag==2)
631  selectedContext = context;
632  if (!firstContext)
633  firstContext = context;
634  context->setDirty(false);
635  }
636  }
637  if (selectedContext) return selectedContext;
638  return firstContext;
639  }
