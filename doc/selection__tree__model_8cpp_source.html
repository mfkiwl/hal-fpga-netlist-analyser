181  void SelectionTreeModel::moduleRecursion(SelectionTreeItemModule* modItem)
182  {
183  if (modItem->isRoot()) return;
184  Module* mod = gNetlist->get_module_by_id(modItem->id());
185  if (!mod) return;

187  {
188  SelectionTreeItemModule* subItem = new SelectionTreeItemModule(m->get_id());
189  moduleRecursion(subItem);
190  modItem->addChild(subItem);
191  }
192  for (Gate* g : mod->get_gates() )
193  {
194  modItem->addChild(new SelectionTreeItemGate(g->get_id()));
195  }
196  std::unordered_set internNets = mod->get_internal_nets();
197  std::unordered_set outputNets = mod->get_output_nets();
198  std::unordered_set inputNets = mod->get_input_nets();
199  for (Net* n : mod->get_nets() )
200  {
201  bool netIsChild = false;
202  if (internNets.find(n) != internNets.end())
203  {
204  netIsChild = true;
205  }
206  else if (outputNets.find(n) != outputNets.end())
207  {
208  if (!n->is_global_output_net() && n->get_destinations().empty())
209  netIsChild = true;
210  }
211  else if (inputNets.find(n) != inputNets.end())
212  {
213  if (!n->is_global_input_net() && n->get_sources().empty())
214  netIsChild = true;
215  }
216  else
217  netIsChild = true;
