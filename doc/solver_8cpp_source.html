254  std::set<std::tuple<std::string, u16>> inputs;
255  for (const auto& constraint : _constraints)
256  {
257  if (constraint.is_assignment())
258  {
259  for (const auto& node : constraint.get_assignment().get()->first.get_nodes())
260  {
261  if (node.is_variable())
262  {
263  inputs.insert(std::make_tuple(node.variable, node.size));
264  }
265  }
266  }
267  else
268  {
269  for (const auto& node : constraint.get_function().get()->get_nodes())
270  {
271  if (node.is_variable())
272  {
273  inputs.insert(std::make_tuple(node.variable, node.size));
274  }
275  }
276  }
277  }
278 
279  return std::accumulate(inputs.begin(), inputs.end(), std::string(), [](auto accumulator, auto entry) -> std::string {
280  return accumulator + "(declare-fun " + std::get<0>(entry) + " () (_ BitVec " + std::to_string(std::get<1>(entry)) + "))\n";
281  });
282  };
283 
288  auto translate_constraints = [](const std::vector<Constraint>& _constraints) -> Result<std::string> {
289  return std::accumulate(_constraints.cbegin(), _constraints.cend(), Result<std::string>::Ok({}), [](auto accumulator, const auto& constraint) -> Result<std::string> {
290  // (1) short-hand termination in case accumulator is an error
291  if (accumulator.is_error())
292  {
293  return ERR(accumulator.get_error());
294  }
295 
296  if (constraint.is_assignment())
297  {
298  const auto assignment = constraint.get_assignment().get();
299  auto lhs = Translator::translate_to_smt2(assignment->first);
300  auto rhs = Translator::translate_to_smt2(assignment->second);
301  if (lhs.is_error())

303  return ERR_APPEND(lhs.get_error(), "could not translate constraint to SMT-LIB v2: '" + constraint.to_string() + "'");
304  }
305  else if (rhs.is_error())
306  {
307  return ERR_APPEND(rhs.get_error(), "could not translate constraint to SMT-LIB v2: '" + constraint.to_string() + "'");
308  }
309  else
310  {
311  return OK(accumulator.get() + "(assert (= " + lhs.get() + " " + rhs.get() + "))\n");
312  }
313  }
314  else
315  {
316  auto lhs = Translator::translate_to_smt2(*constraint.get_function().get());
317  if (lhs.is_ok())

320  }
321  return ERR_APPEND(lhs.get_error(), "could not translate constraint to SMT-LIB v2: '" + constraint.to_string() + "'");
322  }
323  });
324  };
