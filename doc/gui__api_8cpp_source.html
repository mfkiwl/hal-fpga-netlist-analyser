448  int GuiApiClasses::View::isolateInNew(std::vector<Module*> modules, std::vector<Gate*> gates)
449  {
450  //check if the inputs are valid
451  for(Module* mod : modules)
452  {
453  if (mod == nullptr)
454  {
455  log_warning("gui","Null values not allowed in module argument");
456  return 0;
457  }
458  }
459 
460  for(Gate* gate : gates)
461  {
462  if (gate == nullptr)
463  {
464  log_warning("gui","Null values not allowed in gate argument");
465  return 0;
466  }
467  }
468  QString name;
469 
470  bool isModuleExclusive = false;
471 
472  //make sure that modules and gates are not empty
473  if(modules.empty() && gates.empty())
474  return 0;
475 
476  //Check if view should be bound exclusively to module
477  if(modules.size() == 1 && gates.empty() && modules[0]){
478  name = QString::fromStdString(modules[0]->get_name()) + QString(" (ID: %1)").arg(modules[0]->get_id());
479  isModuleExclusive = true;
480  //If the view already exists then return existing id
481  if(gGraphContextManager->contextWithNameExists(name)){
482  for(GraphContext* ctx : gGraphContextManager->getContexts()){
483  if(ctx->name() == name){
484  return ctx->id();
485  }
486  }
487  return 0;
488  }
489  }
490  else
491  {
492  //Get the number which has to be appended to name
493  name = gGraphContextManager->nextViewName("Isolated View");
494  }
495  GuiApiClasses::View::ModuleGateIdPair pair = GuiApiClasses::View::getValidObjects(0, modules, gates);
496 
497  // Unpack return values and check if not empty
498  QSet<u32> moduleIds = pair.moduleIds;
499  QSet<u32> gateIds = pair.gateIds;
500 
501  if(moduleIds.isEmpty() && gateIds.isEmpty())
502  return 0;
503 
504  UserActionCompound* act = new UserActionCompound;
505  act->setUseCreatedObject();
506  act->addAction(new ActionCreateObject(UserActionObjectType::Context, name));
507  act->addAction(new ActionAddItemsToObject(moduleIds, gateIds));
508  UserActionManager::instance()->executeActionBlockThread(act);
509 
510  if (isModuleExclusive){
511  GraphContext* context = gGraphContextManager->getContextById(act->object().id());
512  context->setDirty(false);
513  context->setExclusiveModuleId(modules[0]->get_id());
514  }
515  return act->object().id();
516  }

795  {
796  Module* topModule = gNetlist->get_top_module();
797  //copy to prevent inplace operations
798  std::vector<Module*> modules = mods;
799  std::vector<Gate*> gates = gats;
800 
801  //set to store already existing mods and gates
802  QSet<u32> existingModules;
803  QSet<u32> existingGates;
804 
805  QSet<u32> Parents;
806  QSet<u32> modIds;
807  QSet<u32> gatIds;
808  //0) if its not a new view we have to check and remove all parents which have to be placed
809  if (viewId)
810  {
811  //put topmodule into parents because we dont iterate over it here
812  //Parents.insert(topModule->get_id());
813  std::vector<Module*> validMods;
814  //Add parents from the view modules to Parents
815  for (Module* mod : GuiApiClasses::View::getModules(viewId))
816  {
817  existingModules.insert(mod->get_id());
818  validMods.push_back(mod);
819  Module* itr = mod->get_parent_module();
820  while (itr != nullptr)
821  {
822  if (Parents.contains(itr->get_id()))
823  break;
824  Parents.insert(itr->get_id());
825  itr = itr->get_parent_module();
826  }
827 
828  }
829  //Add parents from gate to parents
830  for (Gate* gate : GuiApiClasses::View::getGates(viewId))
831  {
832  existingGates.insert(gate->get_id());
833  Module* itr = gate->get_module();
834  while (itr != nullptr)
835  {
836  if (Parents.contains(itr->get_id()))
837  break;
838  Parents.insert(itr->get_id());
839  itr = itr->get_parent_module();
840  }
841  }
842  //Delete every parent from the list if submodule is in the view
843  for (Module* mod : modules)
844  {
845  if (Parents.contains(mod->get_id()))
846  continue;
847  validMods.push_back(mod);
848  }
849  modules = validMods;
850  }
851  //get ids of modules
852  for (Module* mod : modules)
853  {
854  modIds.insert(mod->get_id());
855  }
856  //1) sort them by priority in DESCENDING order
857  std::sort(modules.begin(), modules.end(), [](const Module* a, const Module* b) -> bool
858  {
859  //sorting in DESCENDING order to check from lowest to highest priority
860  return a->get_submodule_depth() > b->get_submodule_depth();
861  }
862  );
863  //2) remove id if parent is in set
864  for(Module* mod : modules){
865  //check if top module
866  if(mod == topModule){
867  // only add the top module because it has highest priority
868  QSet<u32> temp;
869  temp.insert(topModule->get_id());
870  modIds = temp;
871  break;
872  }
873  //check if parent is in current set and if so remove current mod
874  Module* iterator = mod->get_parent_module();
875  while(iterator != nullptr){
876  if(modIds.contains(iterator->get_id()))
877  {
878  // parent is already in the set so remove mod and stop traversing parent tree
879  modIds.remove(mod->get_id());
880  break;
881  }
882  iterator = iterator->get_parent_module();
883  }
884  }
885  //3) remove all gates which has its ancestor in modIds
886  //check ancestors until topmodule or found in modIds
887  for (Gate* gate : gates)
888  {
889  Module* itr = gate->get_module();
890  bool shouldInsert = true;
891  while (itr != nullptr)
892  {
893  if (modIds.contains(itr->get_id()))
894  {
895  gatIds.remove(gate->get_id());
896  shouldInsert = false;
897  break;
898  }
899  itr = itr->get_parent_module();
900  }
901  if (shouldInsert)
902  gatIds.insert(gate->get_id());
903  }
904 
905  //remove duplicates
906  if (viewId)
907  {
908  modIds -= existingModules;
909  gatIds -= existingGates;
910  }
911  //create struct to return module and gate ID pairs
912  GuiApiClasses::View::ModuleGateIdPair pair;
913  pair.moduleIds = modIds;
914  pair.gateIds = gatIds;
915 
916  return pair;
917 
918  }
