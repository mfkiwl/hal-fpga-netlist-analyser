273  bool ModulePinsTreeModel::canDropMimeData(const QMimeData *data, Qt::DropAction action, int row, int column, const QModelIndex &parent) const
274  {
275  Q_UNUSED(action)
276  if(!data->formats().contains("pintreemodel/item")) return false;
277 
278  QString type; int id;
279  auto encItem = data->data("pintreemodel/item");
280  QDataStream dataStream(&encItem, QIODevice::ReadOnly);
281  dataStream >> type >> id;
282  auto parentItem = getItemFromIndex(parent);
283  qDebug() << "type: " << type << ", id" << id << ", row: " << row;
284 
285  // construct a "drop-matrix" here, but only 4(5) things are NOT allowed (so check for these):
286  // 1: drop a pin on its OWN parent
287  // 3: drop a pingroup on/between pins
288  // 4: drop an item on itself
289  // 5: drop adjecent index to itself, must be at least 1 item apart
290  if(type == "group")
291  {
292  auto item = mIdToGroupItem[id];
293  if(parentItem)
294  {
295  auto type = getTypeOfItem(parentItem);
296  if(type == itemType::pin || (type == itemType::group && row != -1) || item == parentItem)
297  return false;
298  }
299  else // here, only check for adjacent rows
300  {
301  auto itRow = item->getOwnRow();
302  if(itRow == row || ((itRow+1) == row))
303  return false;
304  }
305  }
306  if(type == "pin")
307  {
308  // perhaps check here that a pin can only be dropped between groups if its own group has size > 1?
309  // otherwise it does not make much sense...perhaps change this check
310  auto item = mIdToPinItem[id];
311  if((!parentItem && item->getParent()->getChildCount() == 1) || (item->getParent() == parentItem && row == -1) || item == parentItem
312  || (parentItem && (getTypeOfItem(parentItem) == itemType::pin)))
313  return false;
314  // case if one wants to drop between pins in same group, check if its not adjacent row (other cases are handled on case above
315  if(item->getParent() == parentItem)
316  {
317  auto itRow = item->getOwnRow();
318  if(itRow == row || ((itRow+1) == row))
319  return false;
320  }
321  }
322  return true;
323  }

548  ActionReorderObject* reorderObj = new ActionReorderObject(desiredIdx);

552  }

566  compAct->addAction(reorderActHack);
567  compAct->addAction(addAct);
568  compAct->addAction(reorderAct);
569  compAct->exec();
570  }

572  removeItem(droppedPin);
573  insertItem(droppedPin, onDroppedParent, desiredIdx);
574  if(!(oldParent->getChildCount())){
575  removeItem(oldParent);

577  }
