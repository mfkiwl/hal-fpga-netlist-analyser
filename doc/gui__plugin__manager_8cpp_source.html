279  void GuiPluginTable::populateTable(bool refresh)
280  {
281  QDir userConfigDir(QString::fromStdString(utils::get_user_config_directory()));
282  QMap<QString,GuiPluginEntry*> pluginEntries;
283  mSettings = new QSettings(userConfigDir.absoluteFilePath("plugins.ini"),QSettings::IniFormat);
284 
285  std::set<std::string> loadedPlugins = plugin_manager::get_plugin_names();
286 
287  int nentry = mSettings->beginReadArray("plugins");
288  for (int i=0; i<nentry; i++)
289  {
290  mSettings->setArrayIndex(i);
291  GuiPluginEntry* gpe = new GuiPluginEntry(mSettings);
292  pluginEntries.insert(gpe->mName,gpe);
293  }
294  mSettings->endArray();
295 
296  for (std::filesystem::path plugDir : utils::get_plugin_directories())
297  {
298  QDir dir(QString::fromStdString(plugDir.string()));
299  if (!dir.exists()) continue;
300  for (QFileInfo info : dir.entryInfoList())
301  {
302  if (!info.isFile()) continue;
303  if (!plugin_manager::has_valid_file_extension(info.absoluteFilePath().toStdString())) continue;
304 
305  QString pluginName = info.baseName();
306  GuiPluginEntry* gpe = pluginEntries.value(pluginName);
307  bool needUpdate = true;
308  if (gpe)
309  {
310  needUpdate = gpe->mFileModified != info.lastModified();
311  }
312  else
313  {
314  gpe = new GuiPluginEntry(info);
315  pluginEntries.insert(pluginName,gpe);
316  }
317 
318  BasePluginInterface* bpif = plugin_manager::get_plugin_instance(pluginName.toStdString(), true, true);
319  if (bpif && !refresh)
320  {
321  // plugin already loaded ?
322  bool alreadyLoaded = loadedPlugins.find(pluginName.toStdString())!=loadedPlugins.end();
323  gpe->updateFromLoaded(bpif, alreadyLoaded);
324  log_debug("gui", "GuiPluginManager: '{}' already loaded.", pluginName.toStdString());
325  continue;
326  }
327  else if (gpe->requestLoad())
328  {
329  // user requested this plugin explicitly in previous session
330  bpif = load(pluginName, info.absoluteFilePath());
331  if (!bpif)
332  {
333  gpe->mState = GuiPluginEntry::NotAPlugin; // however, load failed
334  log_warning("gui", "GuiPluginManager: loading of requested plugin '{}' failed.", pluginName.toStdString());
335  }
336  else
337  {
338  gpe->updateFromLoaded(bpif, true, info.lastModified());
339  log_info("gui", "GuiPluginManager: '{}' loaded upon user request.", pluginName.toStdString());
340  }
341  continue;
342  }
343 
344  if (!needUpdate)
345  {
346  log_debug("gui", "GuiPluginManager: '{}' not modified, no update required.", pluginName.toStdString());
347  // known plugin, load only when needed
348  if (gpe->isPlugin())
349  gpe->mState = GuiPluginEntry::NotLoaded;
350  continue;
351  }
352 
353  // plugin updated or not known yet, try dummy load to retrieve information
354  bpif = load(pluginName, info.absoluteFilePath());
355  if (!bpif)
356  {
357  log_warning("gui", "GuiPluginManager: loading of '{}' failed, is it really a HAL plugin?", pluginName.toStdString());
358  gpe->mState = GuiPluginEntry::NotAPlugin;
359  continue;
360  }
361 
362  if (bpif->get_name().empty() || bpif->get_version().empty())
363  {
364  log_warning("gui", "GuiPluginManager: '{}' has empty name or version, plugin ignored.", pluginName.toStdString());
365  gpe->mState = GuiPluginEntry::NotAPlugin;
366  }
367  else
368  {
369  log_info("gui", "GuiPluginManager: '{}' loaded to retrieve information about plugin features.", pluginName.toStdString());
370  gpe->updateFromLoaded(bpif, false, info.lastModified()); // load success
371  }
372  }
373  }
374 
375  // update entry if plugin loaded in meantime as dependency
376  for (std::string loadedPluginName: plugin_manager::get_plugin_names())
377  {
378  GuiPluginEntry* gpe = pluginEntries.value(QString::fromStdString(loadedPluginName));
379  if (!gpe || gpe->isLoaded()) continue;
380  BasePluginInterface* bpif = plugin_manager::get_plugin_instance(loadedPluginName,false,true);
381  if (!bpif) continue;
382  QFileInfo info(gpe->mFilePath);
383  gpe->updateFromLoaded(bpif,false,info.exists()?info.lastModified():QDateTime());
384  }
385 
386  mEntries.clear();
387  mAvoid.clear();
388  mLookup.clear();
389  for (auto it = pluginEntries.constBegin(); it != pluginEntries.constEnd(); ++it)
390  {
391  GuiPluginEntry* gpe = it.value();
392  if (gpe->isPlugin())
393  {
394  mLookup.insert(gpe->mName,mEntries.size());
395  mEntries.append(gpe);
396  }
397  else
398  {
399  mAvoid.append(gpe);
400  }
401  }
402  persist();
403  }

542  void GuiPluginTable::handleButtonPressed(const QModelIndex& buttonIndex)
543  {

547  if (irow >= mEntries.size()) return;
548  GuiPluginEntry* gpe = mEntries.at(irow);
549  switch (buttonIndex.column())
550  {
551  case 7:
552  if (gpe->mName == "hal_gui") return;
553  if (gpe->isLoaded())
554  Q_EMIT triggerInvokeGui(gpe->mName);
555  return;
556  case 8:
557  if (!gpe->mCliOptions.isEmpty())
558  Q_EMIT showCliOptions(gpe->mName, gpe->mCliOptions);
559  return;
560  }
561  if (gpe->mName == "hal_gui") return;
562 
563  bool success = false;
564  if (gpe->isLoaded())
565  {
566  QStringList need = neededBy(gpe->mName);
567  if (need.isEmpty())
568  {
569  success = plugin_manager::unload(gpe->mName.toStdString());
570  if (success)
571  gpe->mState = GuiPluginEntry::NotLoaded;
572  else
573  QMessageBox::warning(qApp->activeWindow(),"Unload failed",
574  QString("Unload of plugin %1\nrefused by plugin_manager")
575  .arg(gpe->mName));
576  }
577  else
578  {
579  QMessageBox::warning(qApp->activeWindow(),"Unload failed",
580  QString("Cannot unload plugin %1\nwhich is required by %2")
581  .arg(gpe->mName).arg(need.join(',')));
582  }
583  }
584  else
585  {
586  if (plugin_manager::load(gpe->mName.toStdString(),gpe->mFilePath.toStdString()))
587  {
588  success = true;
589  gpe->mState = GuiPluginEntry::UserLoad;
590  }
591  else
592  QMessageBox::warning(qApp->activeWindow(),"Load failed",
593  QString("Could not load plugin %1\nplease check plugin_manager log")
594  .arg(gpe->mName));
595  }
596  if (success)
597  {
598  Q_EMIT dataChanged(index(irow,0),index(irow,columnCount()-1));
599  persist();
600  }

702  case 7: return mGuiExtensions;
703  case 8: return mCliOptions;
704  case 9: return isLoaded() ? "LOADED" : "-";
705  case 10: return (int) mState;

835  }
836  break;
837  default:

839  break;
840  }
