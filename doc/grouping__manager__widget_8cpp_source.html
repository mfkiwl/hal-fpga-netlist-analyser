306  void GroupingManagerWidget::newGroupingSuccOrPred(int maxDepth, bool succ, const GraphicsItem* item)
307  {
308  QSet<u32> mods, gats, nets;
309  ToolboxNode tbn;
310  if (item)
311  tbn = ToolboxNode(item);
312  ToolboxModuleHash tmh(tbn.mNode);
313  QVector<Net*> todoNet;
314  QSet<Node> handledBox;
315  QSet<Net*> handledNet;
316  todoNet.append(QVector<Net*>::fromStdVector(succ ? tbn.outputNets() : tbn.inputNets()));
317  handledBox.insert(tbn.mNode);
318  for (int loop = 0; !maxDepth || loop < maxDepth; loop++)
319  {
320  if (todoNet.isEmpty())
321  break;
322  QVector<Net*> nextRound;
323  for (Net* n : todoNet)
324  {
325  if (handledNet.contains(n))
326  continue;
327  handledNet.insert(n);
328  nets.insert(n->get_id());
329  for (Endpoint* ep : (succ ? n->get_destinations() : n->get_sources()))
330  {
331  ToolboxNode nextNode(ep, &tmh);
332  if (handledBox.contains(nextNode.mNode))
333  continue;
334  handledBox.insert(nextNode.mNode);
335  switch (nextNode.mNode.type())
336  {
337  case Node::Module:
338  mods.insert(nextNode.mNode.id());
339  break;
340  case Node::Gate:
341  gats.insert(nextNode.mNode.id());
342  break;
343  default:
344  continue;
345  }
346  nextRound.append(QVector<Net*>::fromStdVector(succ ? nextNode.outputNets() : nextNode.inputNets()));
347  }
348  }
349  todoNet = nextRound;
350  }
351 
352  UserActionCompound* act = new UserActionCompound;
353  act->setUseCreatedObject();
354  act->addAction(new ActionCreateObject(UserActionObjectType::Grouping, tbn.mName + (succ ? " successor" : " predecessor")));
355  act->addAction(new ActionAddItemsToObject(mods, gats, nets));
356  act->exec();
357  }

369  void GroupingManagerWidget::newGroupingByDistance(int maxDepth, bool succ, const GraphicsItem* item)
370  {
371  ToolboxNode tbn;
372  if (item)
373  tbn = ToolboxNode(item);
374  ToolboxModuleHash tmh(tbn.mNode);
375  QVector<Net*> thisDst;
376  QVector<Net*> nextDst;
377  QSet<Node> handledNode;
378  QSet<Net*> handledNet;
379  nextDst.append(QVector<Net*>::fromStdVector(succ ? tbn.outputNets() : tbn.inputNets()));
380  handledNode.insert(tbn.mNode);
381  for (int iround = 1; iround <= maxDepth; iround++)
382  {
383  if (nextDst.isEmpty())
384  break;
385  thisDst = nextDst;
386  nextDst.clear();
387  QSet<u32> mods, gats, nets;
388  for (Net* n : thisDst)
389  {
390  if (handledNet.contains(n))
391  continue;
392  handledNet.insert(n);
393  nets.insert(n->get_id());
394  for (Endpoint* ep : succ ? n->get_destinations() : n->get_sources())
395  {
396  ToolboxNode nextNode(ep, &tmh);
397  if (handledNode.contains(nextNode.mNode))
398  continue;
399  handledNode.insert(nextNode.mNode);
400  switch (nextNode.mNode.type())
401  {
402  case Node::Module:
403  mods.insert(nextNode.mNode.id());
404  break;
405  case Node::Gate:
406  gats.insert(nextNode.mNode.id());
407  break;
408  default:
409  continue;
410  }
411  nextDst.append(QVector<Net*>::fromStdVector(succ ? nextNode.outputNets() : nextNode.inputNets()));
412  }
413  }
414  if (!mods.isEmpty() || !nets.isEmpty() || !gats.isEmpty())
415  {
416  UserActionCompound* act = new UserActionCompound;
417  act->setUseCreatedObject();
418  act->addAction(new ActionCreateObject(UserActionObjectType::Grouping, QString("%1 step to %2").arg(iround).arg(tbn.mName)));
